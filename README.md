        function convertToPerfectFont(text, styleIndex) {

            var dict = fontDicts[styleIndex];

            var result = "";

            for (var i = 0; i < text.length; i++) {

                var char = text[i];

                result += dict[char] || char; 

            }

            return result;

        }

        function startGenerating() {

            var inputName = document.getElementById("customName").value.trim();

            

            if(inputName === "") {

                alert("মুসা ভাই, আগে বক্সে আপনার গেমের নিকনেম ইংরেজিতে লিখুন!");

            document.getElementById("nameList").innerHTML = "";

            setTimeout(function() {

                document.getElementById("loadingCircle").style.display = "none";

                generateAiNames(inputName);

            }, 600);

        }

        function generateAiNames(name) {

            var container = document.getElementById("nameList");

            var htmlContent = "";

            for (var i = 0; i < leftSymbols.length; i++) {

                for (var j = 0; j < fontDicts.length; j++) {

                    

                    var left = leftSymbols[i];

                    var right = rightSymbols[i]; 

                    

                    var styledName = convertToPerfectFont(name, j);

                    var finalName = left + styledName + right;

                    var nameWithPc = finalName + " " + myPcLogo;

                    htmlContent += '<div class="name-card">' +

                                        '' +

                                   '';

                }

            }

            container.innerHTML = htmlContent;

        }

        function copyText(text, message) {

            navigator.clipboard.writeText(text).then(function() {

                alert(message + "\n👉 " + text);

            }).catch(function(err) {

                var textArea = document.createElement("textarea");

                textArea.value = text;

                document.body.appendChild(textArea);

                textArea.select();

                document.execCommand("copy");

                document.body.removeChild(textArea);

                alert(message + "\n👉 " + text);

            });
